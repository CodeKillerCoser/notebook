1完整代码

use std::io;
use rand;
use std::cmp::Ordering;

fn main() {
    // 1. 生成随机数
    let secret_number = rand::random_range(1..=100);

    println!("Guess the number!");

    // 2. 无限循环，直到猜对
    loop {
        println!("Please input your guess.");

        // 3. 读取用户输入
        let mut guess = String::new();

        // 4. 用 match 处理输入错误
        match io::stdin().read_line(&mut guess) {
            Ok(_) => {
                println!("You guessed: {}", guess);

                // 5. 解析数字，匹配比较结果
                match guess.trim().parse::<i32>() {
                    Ok(guess_num) => {
                        match guess_num.cmp(&secret_number) {
                            Ordering::Less => println!("Too small!"),
                            Ordering::Greater => println!("Too big!"),
                            Ordering::Equal => {
                                println!("You win!");
                                break;
                            }
                        }
                    },
                    _ => println!("Try again!"),  // 解析失败
                }
            },
            Err(e) => {
                println!("Error: {}", e);
                continue;
            }
        }
    }
}

2涉及的知识点

1. 模块导入

use std::io;            // 标准库 I/O
use rand;               // 外部 crate：随机数
use std::cmp::Ordering; // 比较结果枚举

2. 随机数生成（rand crate 0.10+）

let secret_number = rand::random_range(1..=100);

详见 rand crate 版本兼容 文档。

3. 控制台输入

let mut guess = String::new();
io::stdin().read_line(&mut guess)

详见 控制台 I/O 文档。

4. 错误处理

两层错误处理：

// 外层：I/O 错误
match io::stdin().read_line(&mut guess) {
    Ok(_) => { /* 继续 */ },
    Err(e) => { /* 处理 I/O 错误 */ }
}

// 内层：数字解析错误
match guess.trim().parse::<i32>() {
    Ok(num) => { /* 比较 */ },
    _ => { /* 输入不是数字 */ }
}

5. 字符串转数字

guess.trim().parse::<i32>()
  • .trim() — 去掉 read_line 包含的换行符
  • .parse::<i32>() — 解析为 32 位整数

6. 值比较

guess_num.cmp(&secret_number)  // 返回 Ordering 枚举

用 cmp 方法而非 ==，配合 match 处理三种比较结果。

7. 无限循环

loop {
    // ...
    break;  // 猜对时退出
}

详见 流程控制 文档。

3改进方向

增加猜测次数统计

let mut attempts = 0;

loop {
    attempts += 1;
    // ...
    // 猜对后打印
    println!("You win! Attempts: {}", attempts);
}

增加输入验证范围

match guess_num {
    1..=100 => {
        match guess_num.cmp(&secret_number) { /* ... */ }
    },
    _ => println!("Please enter a number between 1 and 100"),
}

支持输入 "quit" 退出

let input = guess.trim();
if input == "quit" || input == "exit" {
    println!("Goodbye!");
    break;
}
